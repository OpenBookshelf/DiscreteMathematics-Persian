\SUBSECTION{سری}

\begin{DEFINITION}
    اگر 
    $a_n$
    یک دنباله باشد به
    \[\sum_{i=0}^{\infty}a_n\]
    یک 
    \FOCUSEDON{سری بی نهایت}
    یا به اختصار, سری می‌گوییم
\end{DEFINITION}
مسئله مورد اهمیت برای ما در مورد سری ها همگرا یا واگرا بودن آن‌ها است.
\begin{DEFINITION}
    اگر
    $a_n$
    یک دنباله باشد,
    به دنباله
    $S_n=a_n+a_{n-1}+...a_0$
    \FOCUSEDON{مجموع جزئی $n$ ام}
    می گوییم.
\end{DEFINITION}


\begin{THEOREM}
    اگر دنباله 
    $S_n$
    همگرا باشد, در این صورت می‌گوییم که سری همگرا است.
\end{THEOREM}

\begin{PROBLEM}
    همگرایی یا واگرایی سری هندسی را بررسی کنید
    (سری هندسی, سری بی‌نهایت دنباله هندسی است)
    \[\sum_{n=1}^{\infty}ar^{n-1}\]
    \SOLUTION{
        ابتدا $S_n$ را می‌نویسیم:
        \[S_n=\sum_{i=1}^{n}ar^{i-1}\]
        \[S_n=\dfrac{a(1-r^n)}{1-r}\]
        حال کافی است مقدار
        $\lim\limits_{n\to \infty} S_n$
        را بررسی کنیم:
        \begin{itemize}
            \item $r\geq 1$
            در این صورت حد موجود نیست و سری واگرا است.
            \item $r < 1$
            از محاسبه حد داریم:
            $$\lim_{n\to \infty} S_n=\dfrac{a}{1-r}$$
            پس در این حالت حد موجود و سری همگرا است.
        \end{itemize}
    }   
\end{PROBLEM}

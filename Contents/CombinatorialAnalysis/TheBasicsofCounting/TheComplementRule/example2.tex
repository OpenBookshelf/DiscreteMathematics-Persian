\begin{PROBLEM}
    \p
    تعداد کلمه‌های عبور با هر یک از شرایط زیر را بیابید.

    \begin{enumerate}
        \item 
        هر کلمه عبور شامل ۶ رقم یا حرف انگلیسی بزرگ
        \lr{(Capital)}
        باشد.
        \SOLUTION{
            \p
            طبق اصل ضرب :
            $$36^6$$
        }

        \item 
        هر کلمه عبور شامل ۶ تا ۸ رقم یا حرف انگلیسی بزرگ باشد.
        \SOLUTION{
            \p
            طبق اصل جمع در ادامه پاسخ قبلی :
            $$36^6 + 36^7 + 36^8$$
        }   
        
        \item 
        هر کلمه عبور شامل ۶ تا ۸ رقم یا حرف انگلیسی بزرگ باشد. هر کلمه عبور باید حداقل
        یک حرف داشته باشد.
        \SOLUTION{
            \p
            طبق اصل متمم در ادامه پاسخ قبلی :
            $$(36^6 + 36^7 + 36^8) - (10^6 + 10^7 + 10^8)$$
            که عبارت کاسته شده از جواب نشان‌دهنده‌ی تعداد حالات بدون حرف است.
        }

        \item 
        هر کلمه عبور شامل ۶ تا ۸ رقم یا حرف انگلیسی بزرگ باشد. هر کلمه عبور باید حداقل
        یک حرف و یک رقم داشته باشد.
        \SOLUTION{
            \p
            طبق اصل جمع در ادامه پاسخ قبلی :
            $$(36^6 + 36^7 + 36^8) - ((10^6 + 10^7 + 10^8) + (26^6 + 26^7 + 26^8))$$
        }

        \item 
        هر کلمه عبور شامل ۶ تا ۸ رقم یا حرف انگلیسی بزرگ باشد. هر کلمه عبور باید حداقل
        یک حرف و یک رقم داشته باشد مگر آنکه فقط از حروف کاملا متمایز تشکیل شده باشد و شامل
        ۶ کاراکتر باشد.
        \SOLUTION{
            \p
            طبق اصل جمع در ادامه پاسخ قبلی :
            $$(36^6 + 36^7 + 36^8) - ((10^6 + 10^7 + 10^8) +$$
            $$(26^6 + 26^7 + 26^8)) + (26 \times 25 \times 24 \times 23 \times 22 \times 21)$$
        }

        \item 
        هر کلمه عبور شامل ۶ تا ۸ رقم یا حرف انگلیسی بزرگ باشد. هر کلمه عبور باید حداقل
        یک حرف و یک رقم داشته باشد مگر آنکه فقط از حروف کاملا متمایز تشکیل شده باشد.
        \SOLUTION{
            \p
            برای اضافه کردن کلمات عبور ۷ یا ۸ کاراکتری بدون رقم به پاسخ قبل،
            می‌توانیم یک کاراکتر «عدم وجود» برای دو کاراکتر آخر متصور شویم. توجه کنید که
            درصورت عدم وجود کاراکتر هفتم، کاراکتر هشتم ملزم به عدم وجود است. برای اعمال این شرط،
            از اصل جمع بر روی دو حالت وجود یا عدم وجود کاراکتر هفتم استفاده می‌کنیم :
            $$(36^6 + 36^7 + 36^8) - ((10^6 + 10^7 + 10^8) +$$
            $$(26^6 + 26^7 + 26^8)) + (26 \times 25 \times 24 \times 23 \times 22 \times 21 \times (20 \times 20 + 1))$$
            سه مقدار ۲۰، ۲۰ و ۱ به پاسخ اضافه شده است که به ترتیب عبارت‌اند از :
            ۱. کاراکتر هفتم یکی از حروف باقی‌مانده باشد.
            ۲. کاراکتر هشتم یکی از حروف باقیمانده یا «عدم وجود» باشد.
            ۳. کاراکتر هفتم «عدم وجود» باشد.
        }
    \end{enumerate}
\end{PROBLEM}

\begin{PROBLEM}
  \p
  ضریب عبارت 
  $x^{12}y^{13}$
  را در بسط زیر بدست آورید.
  \begin{enumerate}
    \item 
    $(x+y)^{25}$

    \SOLUTION[\REF{R} پاسخ]{
      \p
      طبق قضیه ضرایب دو جمله‌ای، ضریب عبارت
      $x^{12}y^{13}$
      برابر است با:
        $$\binom{25}{13} = \frac{25!}{13!12!} = 5200300$$
    }

    \item 
    $(2x-3y)^{25}$

    \SOLUTION[\REF{R} پاسخ]{
      \p
      با توجه به این که ۲۵، عددی صحیح و نا‌منفی است، می‌توانیم از قضیه ضرایب دو جمله‌ای استفاده کنیم؛ طبق این قضیه داریم:
        $$((2x) + (-3y))^{25} = \sum\limits_{j=0}^{25} \binom{25}{j} (2x)^{25-j} (-3y)^j$$
        $$= \sum\limits_{j=0}^{25} \binom{25}{j} 2^{25-j} (-3)^j (x)^{25-j} (y)^j$$
      بنابراین، ضریب عبارت
      $x^{12}y^{13}$
      برابر است با:
        $$\binom{25}{13} 2^{12} (-3)^{13} = \frac{-25!}{13!12!} 2^{12} 3^{13}$$
    }
  \end{enumerate}
\end{PROBLEM}

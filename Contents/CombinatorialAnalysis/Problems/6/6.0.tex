    \p
    فرض کنید 
	$a_k$
	كه
	$0 \leq k \leq 2n$،
	 ضریب 
	 $x^k$ در بسط $(1 + x + x^2)^n$ باشد.
	\begin{enumerate}
        \item
    	به ازای هر $k$
		كه
		$0 \leq k \leq n$،
		ثابت کنید:
		 $$a_{n-k} = a_{n+k}$$
    	\item
    	ثابت کنید:
        $$a_0a_1 - a_1a_2 + a_2a_3 - ... - a_{2n-1}a_{2n} = 0$$
    	\item
    	از تساوی
    	$$(1 + x + x^2)(1 - x + x^2) = 1 + x^2 + x^4$$
    	استفاده کرده و ثابت کنید:
    	$$a_0^2 - a_1^2 + a_2^2 - ... + (-1)^{n - 1}a_{n - 1}^2 = \frac{1}{2}(a_n + (-1)^{n + 1}a_n^2)$$
    	
	\end{enumerate}

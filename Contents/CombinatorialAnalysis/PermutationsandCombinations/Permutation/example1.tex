\begin{PROBLEM}
    \p
    در یک آزمون ۱۰۰ نفر شرکت کرده‌اند.

    \begin{enumerate}
        \item 
        به چند روش  می‌توان برنده‌های اول، دوم و سوم را اعلام کرد؟
        \SOLUTION[\REF{R} پاسخ]{
            $$P(100,3) = 100\times 99\times 98\newline = 970200$$
        }

        \item 
        اگر فرض کنیم نفر سوم مشخص شده است، به چند روش می‌توان برنده‌های اول و دوم را اعلام کرد؟
        \SOLUTION{
            $$P(99,2) = 99\times 98 = 9702$$
        }
    \end{enumerate}
\end{PROBLEM}

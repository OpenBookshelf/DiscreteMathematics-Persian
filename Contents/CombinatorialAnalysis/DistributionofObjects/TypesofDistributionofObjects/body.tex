\begin{PROBLEM}[حالات مختلف توزیع اشیا]
  \begin{enumerate}
      \item 
        به چند طریق می‌توان ۷ توپ نامتمایز را در ۴ جعبه نامتمایز با ظرفیت نامحدود، قرار داد؟
        \NOTE{این مسئله ترتیب با اشیا و جعبه‌های نامتمایز است.}

        \SOLUTION{
          \p
          به ۹ روش می‌توان اینکار را انجام داد:
              $$6$$
              $$5,1$$
              $$4,2$$
              $$4,1,1$$
              $$3,3$$
              $$3,2,1$$
              $$3,1,1,1$$
              $$2,2,2$$
              $$2,2,1,1$$
          در پاسخ بالا، هر سطر نمیانده یکی از حالات توزیع است که در هر سطر، هر عدد گویای تعداد توپ‌ها در هر یک از جعبه‌هاست.
        }

      \item 
        اگر بخواهیم ۱۰ توپ که شماره‌گذاری شده‌اند را در ۳ جعبه شماره‌گذاری شده که هر کدام ۲ تا توپ ظرفیت داشته باشند قرار دهیم، به چند حالت می‌توان این کار را انجام داد؟
        \NOTE{این مسئله ترتیب با اشیا و جعبه‌های متمایز است.}

        \SOLUTION{
            \p
            جعبه‌ی چهارمی برای توپ‌های باقیمانده درنظر می‌گیریم. حال طبق ترکیب چندگانه، تعداد روش‌های
            این تقسیم‌بندی برابر است با:
            $${10 \choose 2,2,2,4} = 420$$
        }

      \item 
        به چند روش می‌توان ۱۰ توپ نامتمایز را در ۸ جعبه شماره‌گذاری شده قرار داد؟
        \NOTE{مسئله فوق، ترتیب با اشیا نامتمایز و جعبه‌های متمایز است که آن را با نام معادله سیاله نیز می‌شناسند،
        درحالی که حالت خاصی از معادله سیاله است. در ادامه، بیشتر با آن آشنا خواهید شد.}
        
        \SOLUTION{
          \p
          فرض کنید می‌خواهیم ۱۰ توپ و ۷ مداد را به ترتیب بچینیم و
          درنهایت، توپ‌های میان هر دو مداد، قبل از اولین مداد و یا بعد از آخرین مداد، هر کدام را درون یک جعبه قرار دهیم.
          ترتیب جعبه‌ها با ترتیب فضای بین مداد‌ها هم‌ارز خواهد بود.
          بنابراین لازم است تعداد جایگشت‌های ۱۰ توپ نامتمایز و ۷ مداد نامتمایز در کنار هم را بشماریم.
          طبق آنچه در جایگشت با اعضای تکراری بیان شد، این تعداد برابر است با:
          $$\frac{17!}{10!7!} = 19448$$
          \p
          روش بالا را تحت عنوان یک روش کلی برای حل معادله سیاله در بخش مربوطه خواهید دید.
        }

      \item 
        به چند طریق می‌توان ۴ توپ شماره‌گذاری شده را در ۳ جعبه نامتمایز با ظرفیت نامحدود، قرار داد؟
        \NOTE{این مسئله ترتیب با اشیا متمایز و جعبه‌های نامتمایز است. پاسخ کلی این مسئله تحت عنوان عدد دوم استرلینگ نیز شناخته می‌شود.}
        
        \SOLUTION{
          \p
          مسئله را بر روی تعداد جعبه‌های خالی حالت‌بندی می‌کنیم:
          \begin{enumerate}
              \item 
              اگر هیچ جعبه‌ای خالی نباشد، لازم است توپ‌ها را به یک دسته دو تایی و دو دسته‌ی تکی تقسیم کنیم.
              طبق ترکیب چندگانه می‌توانیم تعداد راه‌های انجام این کار را محاسبه کنیم.
              توجه کنید که جعبه‌ها نامتمایزاند، پس چون دو جعبه با تعداد اعضای مساوی داریم،
              لازم است حاصل ترکیب را بر تعداد جایگشت‌های آن‌ها تقسیم کنیم:
              $${4 \choose 2,1,1} / 2! = 6$$

              \item 
              اگر دقیقا یک جعبه خالی بماند، یا توپ‌ها به دو دسته دوتایی تقسیم می‌شوند، یا به یک دسته سه‌تایی و یک دسته تکی.
              طبق ترکیب چندگانه و اصل جمع، تعداد راه‌های انجام این کار برابر است با:
              $${4 \choose 3,1} + {4 \choose 2,2} / 2! = 7$$

              \item 
              اگر دقیقا دو جعبه خالی باشد، یعنی ناچاریم همه توپ‌ها را درون جعبه آخر قرار دهیم:
              $${4 \choose 4} = 1$$
          \end{enumerate}
          \p
          طبق اصل جمع، پاسخ مسئله برابر است با:
          $$6 + 7 + 1 = 14$$
        }
  \end{enumerate}
\end{PROBLEM}

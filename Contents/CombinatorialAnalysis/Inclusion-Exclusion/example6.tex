\begin{PROBLEM}
  \p
  به چند روش می‌توان ۵ کار متفاوت را بین ۴ کارمند تقسیم کرد به گونه‌ای که
  هر کارمند دست کم یک کار داشته باشد؟
    
    \SOLUTION{
      \p
      به مسئله به چشم تعداد توابع پوشا از مجموعه ۵ عضوی کار‌ها به
      مجموعه ۴ عضوی کارمند‌ها نگاه کرده و از پاسخ مسئله قبل،
      با 
      $m=5, n=4$
      استفاده می‌کنیم :
        $$ \sum\limits_{i=0}^n (-1)^i \binom{n}{i} (n-i)^m = \sum\limits_{i=0}^4 (-1)^i \binom{4}{i} (4-i)^5 $$
        $$= 1 \times 4^5 - 4 \times 3^5 + 6 \times 2^5 - 4 \times 1^5 + 1 \times 0^5 = 1024 - 972 + 192 - 4 = 240$$
    }
\end{PROBLEM}

\begin{PROBLEM}
	\p
	ثابت کنید که برای هر 
	$n \in N$
	داریم:

	$$z + \frac{1}{z} = 2cos(\alpha) \implies z^n + \frac{1}{z^n} = 2 cos(n \alpha)$$

	\SOLUTION{
		\p
		حکم را با استقرای قوی روی
		$n$
		اثبات می‌کنیم.

		واضح است که حکم به ازای 
		$n = 1$
		صحیح است.
		حال حکم را به ازای 
		$n = 2$
		اثبات می‌کنیم.
		در ابتدا توجه کنید داریم:

		$$z^2 + \frac{1}{z^2} = (z + \frac{1}{z})^2 - 2$$

		که طبق فرض مسئله(
			$z + \frac{1}{z} = 2cos(\alpha)$
		)
		برابر است با:
		$4 cos(\alpha)^2 - 2$
		که این عبارت خود برابر است با:
		$2cos(2 \alpha)$

		پس حکم به ازای 
		$n = 2$
		نیز صحیح است.

		حال فرض کنید حکم به ازای
		$n = 1$
		تا 
		$n = k; k\in N, 2\leq k$
		صحیح است.

		ما حکم را به ازای 
		$ n = k + 1$
		اثبات می‌کنیم:

		$$z^{k + 1} + \frac{1}{z^{k + 1}} = (z + \frac{1}{z}) \times (z^k + \frac{1}{z^k}) - z^{k - 1} - \frac{1}{k - 1}$$

		حال طبق فرض استقرا داریم:
		$$ (z^k + \frac{1}{z^k} = 2 cos(k \alpha)), (z^{k - 1} - \frac{1}{k - 1} = 2 cos((k-1)\alpha))$$
		(توجه کنید که در اینجا به استقرای قوی نیاز داریم چون از حکم مسئله به ازای 
		$k - 1$
		هم استفاده کردیم.)

		و طبق فرض مسئله داریم :
		$z + \frac{1}{z} = 2cos(\alpha)$

		پس در کل داریم:
		$$z^{k + 1} + \frac{1}{z^{k + 1}} = (z + \frac{1}{z}) \times (z^k + \frac{1}{z^k}) - z^{k - 1} - \frac{1}{k - 1}
		= 4cos(\alpha)cos(k \alpha) - 2cos((k - 1)\alpha)$$

		حال طبق قضیه جمع کسینوس ها(
			$cos(x + y) + cos(x - y) = 2cos(x)cos(y)$
		)
		داریم:

		$$4cos(\alpha)cos(k \alpha) - 2cos((k - 1)\alpha) = 2cos((k + 1)\alpha) + 2 cos((k - 1)\alpha) - 2cos((k -1)
		\alpha) = 2 cos((k + 1) \alpha)$$

		پس حکم استقرا به ازای 
		$n = k+1$
		نیز صحیح است.

		پس نتیجه می‌گیریم حکم به ازای هر
		$n \in N$
		صحیحی است.
	}
	% منبع(استراتژی حل مسئله ص270)
\end{PROBLEM}

\begin{PROBLEM}
	\p
	تابع 
	$f(m, n)$
	به این صورت تعریف می‌شود که:
	$$f(1, 1) = 2$$
	$$f(m + 1, n) = f(m, n) + 2(m + n)$$
	$$f(m, n + 1) = f(m, n) + 2(m + n - 1)$$
	حال ثابت کنید به ازای هر 
	$m, n \in N$
	داریم :
	$$f(m, n) = (m + n)^2 - (m + n) - 2n + 2$$

	\SOLUTION{
		\p
		برای حل این سوال ابتدا اثبات می‌کنیم به ازای هر 
		$m \in N$
		داریم: 
		$$f(m, 1) = (m + 1)^2 - m - 1 = m^2 + m$$
		طبق صورت سوال حکم به ازای 
		$m = 1$
		صحیح است، حال فرض کنید حکم به ازای 
		$m = k; k \in N$
		صحیح باشد، ما حکم را به ازای 
		$m = k + 1$
		اثبات می‌کنیم.
		$$f(k + 1, 1) = f(k, 1) + 2(k + 1) = k^2 + k + 2k + 2 = (k+1)^2 + k$$

		پس حکم به ازای 
		$m = k + 1$
		نیز صحیح است، پس حکم به ازای هر 
		$m \in N$
		صحیح است.

		حال می خواهیم حکم سوال را به ازای هر 
		$n, m \in N$
		ثابت کنیم.
		برای این کار روی 
		$n$
		استقرا می‌زنیم.

		حکم به ازای
		$n = 1, m \in N$
		طبق اثبات‌های بالا صحیح است.

		حال فرض کنید حکم به ازای 
		$n = k, k \in N$
		صحیح باشد.
		ما حکم را به ازای 
		$n = k + 1$
		اثبات می‌کنیم:
		$$f(m, k + 1) = f(m, k) + (m + k - 1) = (m + k)^2 - (m + k) - 2k + 2 + m + k + 1 = (m + k + 1)^2 - (m + k + 1) -
				2(k + 1) + 2$$
		پس حکم به ازای 
		$n = k + 1$
		نیز صحیح است، پس حکم به ازای هر 
		$n , m \in N$
		صحیح است.
	}

\end{PROBLEM}

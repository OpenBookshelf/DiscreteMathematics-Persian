\begin{PROBLEM}
	\p
	دو دنباله عدد به  صورت زیر داریم:
	$$x_1, x_2, ..., x_n; x_i \in N$$
	$$y_1, y_2, ..., y_m; y_j \in N$$
	که:
	$$x_1 + x_2 + ... + x_n = y_1 + y_2 + ... + y_m = s$$
	و 
	$s < m \times n$.

	ثابت کنید می‌توانیم چند عدد(حداقل یک عدد از هر دنباله) حذف کنیم به طوری که مجوع اعداد باقی مانده در هر دنباله
	باز هم برابر باشند.(نباید کل اعداد یک دنباله را حذف کنیم)

	\SOLUTION{
		\p
		به به،‌سوالات این بخش واقعا لذت بخشه،‌دم آقامون آرتور انگل گرم!
		(و همچنین آقا یاسر خان احمدی فولادی)

		برگردیم سراغ حل سوال:

		لم۱) 
		$n$
		و
		$m$
		هر دو باید از یک بیشتر باشند چون در غیر این صورت داریم:
		$$s >= max(m, n) >= m \times n$$
		که با فرض مسئله در تناقض است.

		حال حکم مسئله را با استقرا روی 
		$n + m = k$
		اثبات می‌کنیم.

		حکم به ازای 
		$k = 4$
		صحیح است،‌اثبات:

		طبق لم۱ در این حالت داریم:
		$n = m = 2$
		در نتیجه داریم: 
		$ 2 \leq s \leq 3$
		در نتیجه هر دو دنباله باید شامل حداقل یک عدد ۱ باشند که با خط زدن یک ۱ از هر دنباله می‌توان جمع اعداد باقی مانده را 
		برابر کرد و به خواسته‌ی مسئله رسید.

		حال فرض کنید حکم سوال به ازای
		$ k = t; t \in N, 4 \leq t$
		صحیح باشد. ما حکم را به ازای 
		$k = t + 1$
		اثبات می‌کنیم:

		$x_1 , y_1$
		را بزرگترین عدد از هر دنباله در نظر بگیرید. و بدون از دست رفتن کلیت مسئله فرض کنید 
		$y_1 \leq x_1$.
		
		 حال اگر این دو عدد برابر باشند با خط زدن این دو به خواسته‌ی سوال می‌رسیم و حکم اثبات می‌شود در غیر این صورت
		 داریم:

		 $$ (x_1 - y_1) + x_2 + ... + x_n = y_2 + y_3 + ... + y_m = s'$$
		 
		 حال چون داریم 
		 $ \frac{s}{m} \leq y_1 $
		 پس خواهیم داشت:

		 $$s' \leq s - \frac{s}{m} = \frac{s (m - 1)}{m} < \frac{m n (m-1)}{m} = n \times (m-1)$$

		 در نتیجه هر دو شرط مسئله برای این دو دنباله ی جدید بر قرار است و چون 
		 $n + m - 1 = t$
		 پس طبق فرض استقرا روشی مثل روش آلفا وجود دارد که در آن می‌توان تعدادی عدد از این دو دنباله حذف کرد به طوری که مجموع اعداد دو دنباله برابر بماند
		 حال اگر 
		 $(x_1 - y_1)$
		 در این راه خط زده شده بود از هر دو دنباله اولیه
		 $x_1, y_1$
		 را حذف می‌کنیم و همچنین بقیه اعداد را همانطور که در راه آلفا خط می‌خورند خط می‌زنیم.

		 و اگر در راه آلفا
		 $(x_1 - y_1)$
		 خط نخورده بود باقی اعداد را مانند راه آلفا خط می‌زنیم و 
		 $y_1$
		 را به جای اولیه خود باز می‌گردانیم.

		 پس در هر صورت می‌توان تعدادی عدد از دو دنباله حذف کرد به طوری که مجموع اعداد باقی مانده برابر شود.
		 پس حکم استقرا به ازای
		 $k = t + 1$
		 صحیح است پس حکم به ازای هر 
		 $k \in N, 4 \leq k, m + n = k, 2 \leq m, n$
		 صحیح است.
	}
	% منبع(استراتژی حل مسئله)
\end{PROBLEM}

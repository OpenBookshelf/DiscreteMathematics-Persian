\begin{PROBLEM}
	\p

	فرض کنید به ازای 
	$\alpha \in R$
	داریم:
	$$\alpha + \frac{1}{\alpha} \in Z$$
	اثبات کنید به ازای هر 
	$n \in W$
	داریم:
	$$\alpha ^ n + \frac{1}{\alpha ^ n} \in Z$$
	
	\SOLUTION{
		\p
		حکم را با استقرا روی 
		$n$
		اثبات می‌کنیم:

		به ازای 
		$n = 0$
		داریم:
		$$\alpha ^ 0 + \frac{1}{\alpha ^ 0} = 2 \in Z$$
		پس حکم به ازای 
		$n = 0$
		برقرار است.
		طبق فرض سوال حکم به ازای 
		$n = 1$
		نیز برقرار است.

		حال فرض کنید حکم از 
		$n = 0$
		تا 
		$n = k; k \in N$
		برقرار باشد. ما حکم را به ازای 
		$n = k + 1$
		اثبات می‌کنیم:

		طبق فرض استقرا داریم:
		$$ \alpha ^ {k - 1} + \frac{1}{\alpha ^ {k - 1}} \in Z, \alpha ^ k + \frac{1}{\alpha ^ {k}} \in Z$$
		پس:
		$$\alpha ^ {k + 1} + \frac{1}{\alpha ^ {k + 1}} = (\alpha + \frac{1}{\alpha})(\alpha ^ k + \frac{1}{\alpha ^ k})
		- (\alpha ^ {k - 1} + \frac{1}{\alpha ^ {k - 1}}) \ in Z$$

		پس حکم استقرا به ازای
		$n = k + 1$
		اثبات شد پس حکم به ازای هر 
		$n \in W$
		صحیح است.
	}
	%(منبع: روش های حل مسئله)
\end{PROBLEM}

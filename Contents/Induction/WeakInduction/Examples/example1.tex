\begin{PROBLEM}
	\p
	اثبات کنید به ازای هر
	$ n \in N$
	داریم:‌
	$ 1^2 + 2^2 + \dots + n^2 = \frac{n(n+1)(2n+1)}{6}$

	\SOLUTION{
		\p
		$n$
		اثبات می‌کنیم.

		پایه استقرا: حکم به ازای
		$n = 1$
		صحیح است و داریم:
		$1^2 = \frac{1*(1 + 1) * (2 * 1 + 1)}{6} = 1$

		حال فرض کنید حکم به ازای
		$n = k, k \in N$
		صحیح است ما حکم را به ازای
		$n = k + 1$
		اثبات می‌کنیم:

		طبق فرض استقرا داریم:
		$1^2 + 2 ^2 + \dots + k^2 = \frac{k * (k + 1) * ( 2k + 1)}{6}$

		پس داریم:

		$1^2 + 2 ^ 2 + \dots + k ^2 + (k+1)^2 = \frac{k * (k + 1) * ( 2k + 1)}{6} + (k + 1)^2 = \frac{2k^3 + k^2 + 2k^2 + k + 6k^2 + 12k + 6}{6} = \frac{2k^3 + 7k^2 + 13k + 6}{6} = \frac{(k+1)((k +1)+1)(2(k+1)+1)}{6}$

		پس حکم استقرا به ازای
		$n = k + 1$
		هم صحیح است پس حکم به ازای هر
		$n \in N$
		صحیح است.
	}
	% منبع(کتاب زرد ترکیبیات ص۱۳۴)
\end{PROBLEM}

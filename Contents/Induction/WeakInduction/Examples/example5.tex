\begin{PROBLEM}
	\NOTE[right-margin]{توجه کنید برای حل این سوال نیاز به دانستن مفاهیم مقدماتی گراف دارید.}

	\p
	اثبات کنید هر گراف ساده بدون مثلث با 
	$2n, n \in N$
	حداکثر
	$n^2$
	یال دارد.

	\SOLUTION{
		\p
		حکم را با استقرا روی 
		$n$
		اثبات می‌کنیم.

		حکم بدیهتا به ازای 
		$ n = 1$
		و یک گراف ساده با دو راس حداکثر یک یال دارد.

		حال فرض کنید حکم به ازای
		$n = k, k \in N$
		صحیح است. ما حکم را به ازای 
		$n = k + 1$
		اثبات می‌کنیم:

		یک گراف با 
		$2(k+1)$
		راس که مثلث ندارد را در نظر بگیرید و دو راس مانند 
		$u, v$
		که به هم یال دارند را از آن حذف کنید.
		حال طبق فرض استقرا این گراف حداکثر 
		$k^2$
		یال دارد.

		دوباره رئوس 
		$u, v$
		را به گراف اضافه کنید.
		به ازای هر راس دیگر در گراف مانند 
		$x$
		حداکثر یکی از رئوس
		$u$
		یا 
		$v$
		به آن یال دارند چون در غیر این صورت این سه راس یک مثلث می‌سازند که تناقض است چون ما فرض کردیم گراف ما فاقد مثلث است.

		پس دو راس 
		$u, v$
		حداکثر 
		$2k$
		یال به رئوس دیگر دارند و یک یال هم بین خودشان دارند و همچنین حداکثر 
		$k^2$
		یال دیگر میان بقیه رئوس موجود است.

		پس گراف دلخواه ما حداکثر
		$k^2 + 2k + 1 = (k + 1)^2$
		یال خواهد داشت.

		پس حکم استقرا به ازای 
		$n = k + 1$
		نیز صحیح است پس حکم سوال به ازای تمام 
		$n, n \in N$
		صحیح است.
	}
	%منبع: استراتژی حل مسئله صفحه 268
\end{PROBLEM}
